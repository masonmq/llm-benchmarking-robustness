################### SummarySE function
summarySE <- function(data=NULL, measurevar, groupvars=NULL, na.rm=TRUE,
                      conf.interval=.95, .drop=TRUE) {
  library(plyr)
  # New version of length which can handle NA's: if na.rm==T, don't count them
  length2 <- function (x, na.rm=TRUE) {
    if (na.rm) sum(!is.na(x))
    else       length(x)
  }
  # This does the summary. For each group's data frame, return a vector with
  # N, mean, and sd
  datac <- ddply(data, groupvars, .drop=.drop,
                 .fun = function(xx, col) {
                   c(N    = length2(xx[[col]], na.rm=na.rm),
                     mean = mean   (xx[[col]], na.rm=na.rm),
                     sd   = sd     (xx[[col]], na.rm=na.rm)
                   )
                 },
                 measurevar
  )
  # Rename the "mean" column    
  datac <- rename(datac, c("mean" = measurevar))
  datac$se <- datac$sd / sqrt(datac$N)  # Calculate standard error of the mean
  # Confidence interval multiplier for standard error
  # Calculate t-statistic for confidence interval: 
  # e.g., if conf.interval is .95, use .975 (above/below), and use df=N-1
  ciMult <- qt(conf.interval/2 + .5, datac$N-1)
  datac$ci <- datac$se * ciMult
  return(datac)
}

### Select variables
fullID <- d %>%
  dplyr::ungroup() %>%
  dplyr::select('domlow', 'totinc', "hhcode") %>%
  dplyr::filter(!is.na(totinc)) %>%
  dplyr::filter(!is.na(domlow)) %>%
  dplyr::filter(!is.na(hhcode))
colnames(fullID)

######### remove the id column from calculating MD
trimID <- fullID %>%
  dplyr::select(-c(3))

########## Compute Mahalanobis Distance
trimID$Mahalnobis<- mahalanobis(trimID, colMeans(trimID), cov(trimID))

########## Add p values
### k-1 degrees of freedom, where k is the number of variables.
trimID$pvalue <- pchisq(trimID$Mahalnobis, df=1, lower.tail=FALSE)

#### Highlight outliers by creating another column
trimID <- trimID %>%
  mutate(Outlier = case_when(pvalue < 0.001 ~ 1,
                             pvalue >= 0.001 ~ 0))
#### How many outliers are there
table(trimID$Outlier)

### Go the way back, add ID to the outlier dataset
trimID <- trimID %>%
  dplyr::mutate(hhcode = fullID$hhcode)

### Prepare datasets so they would have the same colnames for merging
tomerge <- trimID %>%
  dplyr::select(-c(1:2))

######### Merge two datasets, keeping rows with missings Mahalanobis
